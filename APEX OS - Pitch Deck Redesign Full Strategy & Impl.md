import { chromium } from 'playwright';
import sharp from 'sharp';

const STAGING_URL = process.env.STAGING_URL || 'https://apex-os.vercel.app';

async function captureScreenshots() {
  const browser = await chromium.launch();
  const context = await browser.newContext({
    viewport: { width: 1920, height: 1080 },
    deviceScaleFactor: 2,
  });
  const page = await context.newPage();

  // Screenshot 1: Pill Choice System
  await page.goto(STAGING_URL);

  await page.goto(STAGING_URL);
  await page.waitForSelector('.spectacular-terminal', { timeout: 10000 });
  // Toggle geek mode via BrandingBar
  const geekToggle = await page.$('[data-testid="geek-toggle"]');

  // Screenshot 3: Terminal Interaction
  const terminal = await page.$('.terminal-input');

    await terminal.type('help', { delay: 50 });
    await terminal.press('Enter');
    await page.waitForTimeout(1500);
  }
  await page.screenshot({ path: 'raw/03-terminal-interaction.png' });

  await browser.close();

  // Post-process all screenshots
  const files = [
    'raw/01-pill-choice-system.png',
    'raw/02-terminal-geek-mode.png',
    'raw/03-terminal-interaction.png',
  ];

      .composite([{
        input: Buffer.from(
          `<svg><rect width="3840" height="2160" rx="32" ry="32" 
           fill="white"/></svg>`
        ),
        blend: 'dest-in',
      }])
      .png({ quality: 100 })
      .toFile(output);
  }
}

captureScreenshots();
